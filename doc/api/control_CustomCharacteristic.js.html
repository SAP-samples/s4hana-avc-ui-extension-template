sap.ui.define([
	"sap/i2d/lo/lib/vchclf/api/valuation/control/CharacteristicContextContainer",
	"sap/i2d/lo/lib/vchclf/components/valuation/control/characteristic/CharacteristicValueProcessorManager"
], function (CharacteristicContextContainer, CharacteristicValueProcessorManager) {
	"use strict";

	/**
     * Represents a Custom Characteristic within a VC UI Extension view. The custom visualization needs to be placed in aggregation <code>content</code> and can make use of available
	 * UI5 or own controls. Each instance has a <code>CharacteristicValueProcessor</code> attached that becomes involved as soon as a value assignment
	 * should be performed. This processor is decoupled from the actual control representation and implictly used when a generic event handler of 
	 * {@link module:sap/i2d/lo/lib/vchclf/api/valuation/controller/CustomCharacteristicGroupBase}
	 * is called.
     *
     * @public
     * @since 1.0.0
     * @extends module:sap/i2d/lo/lib/vchclf/api/valuation/control/CharacteristicContextContainer
     * 
	 * @module sap/i2d/lo/lib/vchclf/api/valuation/control/CustomCharacteristic
     */
	return CharacteristicContextContainer.extend("sap.i2d.lo.lib.vchclf.api.valuation.control.CustomCharacteristic", { /** @alias module:sap/i2d/lo/lib/vchclf/api/valuation/control/CustomCharacteristic */
		
		renderer: function (oRm, oControl) {
			CharacteristicContextContainer.prototype.getRenderer().render(oRm, oControl);
		},
		
		attachProcessor: function(oCsticData) {
			var oGroupController = this.getGroupController();
			var oGroupClientObject = oGroupController.getCharacteristicGroupClientObject();
			var sTypeId = this.getRegistrationId();
			if (!CharacteristicValueProcessorManager.hasProcessorById(sTypeId)) {
				var oProcessor = CharacteristicValueProcessorManager.createProcessorWithId(sTypeId, oGroupClientObject.getValuationComponent().getView());
				
				var oValuationComp = oGroupClientObject.getValuationComponent();
				var oViewModel = oValuationComp.getViewModel();
				var sPath = oViewModel.getCharacteristicGroupsDataProvider().getCharacteristicPath(oCsticData);
				
				var oGroupBindingContext = oGroupClientObject.getCustomGroupContext();
				oProcessor.init(CharacteristicValueProcessorManager, oGroupBindingContext, sPath, sTypeId, oViewModel, oValuationComp.getModel("valuationSettings"), oValuationComp.getValuationDAO(), oValuationComp.getEmbedderName(), oValuationComp.getGroupRepresentation());
			}
		},
		
		getProcessor: function() {
			var sTypeId = this.getRegistrationId();
			return CharacteristicValueProcessorManager.getProcessorById(sTypeId);
		},
		
		onDataLoaded: function() {
			var oData = this.getBindingContext("view").getObject();
			this.attachProcessor(oData);
		}
	
	});
	
});
